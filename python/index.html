Output(輸出)

基本輸出

            
            """
            # 內建函式 print()
            # 功能：輸出訊息、變數內容或資料值
            # print(參數：變數或資料值)

            設定函式呼叫步驟：
            1. 設定函式名稱
            2. 放一對小括 ()
            3. 放參數
            """

            # print(1個參數)
            # 輸出一個字串
            print('Hello!')

            # print(0個參數)
            # 單純換行
            print()

            # print(2個以上參數，逗號分隔)
            # 參數輸出時，彼此之間空一白
            print('Hello!', 'I am Robot 987.')

            # 可以輸出中文或多國語言文字
            print('哈囉！我是機器人 987。')

輸出不同類別

            
            """
            # 內建函式 type()
            # 功能：回傳變數或資料值的類別
            # type(參數：變數或資料值)
            """

            # 字串 str (string)
            print('123', type('123'))

            # 整數 int (integer)
            print(123, type(123))

            # 浮點數 float
            print(123.0, type(123.0))

            # 布林數 bool (boolean)
            # 例一：硬幣的正反面
            # 例二：電源的開和關
            print(True, type(True))
            print(False, type(False))

字串格式化

            
            """
            字串格式化 (String Formatting)
            1. 小數格式：保留小數位數
            2. 對齊格式：(略，以後再講)

            f-string 設定步驟：
            1. 設定小寫 f
            2. 接一對字串單引號 '' 或雙引號 ""
            3. 視需要放字串常值
            4. 變數或運算式放在一對大括 {} 中
            5. 特定格式要在變數或運算式後接英文冒號 : 再設定
            6. 小數位數例如 .2f 保留小數後 2 位
            """

            # |N = | 是字串常值
            # |n| 是變數
            n = 3
            print(f'N = {n}')

            # |PI = | 是字串常值
            # |pi| 是變數
            # |.4f| 是設定保留小數後 4 位
            pi = 3.1415926
            print(f'PI = {pi:.4f}')

Math(計算)

七種算術運算

            
            # 七種算術運算

            # 輸入兩個整數，存到變數 a 和 b
            a = int(input('a: '))
            b = int(input('b: '))
            # 加(+)、減(-)、乘(*)、除(/)
            a + b
            a - b
            a * b
            a / b
            # 商數(//)、餘數(%)
            # % 其實是模數運算
            a // b
            a % b
            # 指數 (次方)
            a ** b

運算子優先序

            
            # 運算子優先序
            """
            # ex(+).
            # 小括 > 指數 > 乘除、商餘 > 加減
            # 先乘除後加減
            # 指乘除餘加減

            # 整數和浮點數計算結果，會得到浮點數
            例如 5 + 1.0 -> 6.0

            Python 的 3 種除法：
            1. /  : 一般除法，得到浮點數
            2. // : 整數相除，取商數
            3. %  : 整數相除，取餘數
            """

            a = 7 + 2 - 3 * 2
            print('a:', a)

            b = 8 + 4 / 2 - 5 * 2
            print('b:', b)

            c = 8 * 2 + 7 - 9 / 4 * 3 ** 2
            print('c:', c)

            d = (8 + -4) / 2 - 5 * 2
            print('d:', d)

            e = 27 % 5 * 6 ** 2 - 12 // 6
            print('e:', e)

程式錯誤種類

            
            """
            # 題意：將 a 和 b 以算術加法合計

            1. 語法錯誤 Syntax Error  : 不可執行，錯誤類別一律為 SyntaxError
            2. 執行錯誤 Runtime Error : 可以執行，過程顯示錯誤訊息
            3. 語意錯誤 Semantic Error: 可以執行，不會顯示錯誤訊息，結果不符要求

            type(data): 回傳資料的類別
            int('987'): 將數字字串轉成整數 -> integer
            str(12345): 將整數轉成數字字串 -> string
            """

            a = input('a: ))
            b = (input(b: ')
            print(f'a = {a}', type(a))
            print(f'b = {b}', type(b))
            print(a + b)

三倍三顆星

            
            # 三倍三顆星
            """
            # ex(+).
            # i 為 3 的倍數時，輸出 ***，其他值則輸出 *

            1 *
            2 *
            3 ***
            4 *
            5 *
            6 ***
            """

            # range(stop): 0 ~ stop-1
            # range(start, stop): start ~ stop-1
            for i in range(1, 7):
                if False:
                    print(i, '***')
                else:
                    print(i, '*')

星號三角

            
            # 星號三角
            """

            i *

            2 2 **

            4 4 ****
            5 5 *****

            # 倒左直角
            i *
            1 5 *****
            2 4 ****
            3 3 ***

            # 正右直角
            i - *

            2 3 2    **
            3 2 3   ***
            4 1 4  ****
            5 0 5 *****

            # 倒右直角
            i - *
            1 0 5 *****

            3 2 3   ***
            4 3 2    **
            5 4 1     *

            # 金字塔形
            i - *
            1 4 1     *
            2 3 3    ***
            3 2 5   *****
            4 1 7  *******
            5 0 9 *********
            """

            for i in range(1, 6):
                print('*'*i)

等差數列

            range(stop): 0 ~ stop-1

            range(start, stop, step): start ~ stop-1, every step

            # ex(+).
            # 依選項輸出星號三角
            """

            while True:
                print('1.正左直角 2.倒左直角')
                print('3.正右直角 4.倒右直角')
                print('5.金字塔形 q.離開迴圈')
                op = input('> ')
                if op == 'q':
                    break

                n = int(input('n: '))
                if True:
                    for i in range(1, n+1):
                        print(i)
                elif True:
                    for i in range(1, n+1):
                        print(i)
